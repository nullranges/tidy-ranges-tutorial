detail of how we count hits, an ID of NA = no hit

counter <- function(id) sum(!is.na(id)) 
library(tidyr)
null <- x %>% join_overlap_left(boots) %>%
  group_by(iter) %>%
  summarize(n_overlaps=counter(id)) %>%
  as.data.frame() %>%
  complete(iter, fill=list(rate=0)) %>%
  drop_na()
